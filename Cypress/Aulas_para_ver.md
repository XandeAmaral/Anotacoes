function* powers(n) {
  //endless loop to generate
  for (let current = n; ; current *= n) {
    yield current;
  }
}

  // controlling generator
  if (power > 32) {
    break;
  }
  console.log(power);
  // 2
  // 4

  // 16
  // 32
}
